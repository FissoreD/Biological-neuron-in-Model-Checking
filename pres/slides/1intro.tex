\begin{frame}
  \frametitle{The LI\&F neuron}

  \begin{definition}[LI\&F: Leaky Integrate and Fire Model]
    A neuronal network represented by a digraph.
  \end{definition}

  \begin{itemize}
    \item Nodes represent the neurons
    \item The edges (the synaptic connections) can have positive (activators) or negative (inhibitors) weights
    \item A node contain a membrane potential: if its threshold is overcome, a spike is emitted
    \item The leak factor, reduce at each time unit the neuron potential
  \end{itemize}Leaky Integrate and Fire Model


\end{frame}
